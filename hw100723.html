2) Напишите функцию, которая принимает два числа в качестве диапазона и возвращает сумму чисел в указанном диапазоне

function sumInRange(start, end) {

let sum = 0;

for (let i = start; i <= end; i++) {

sum += i;

return sum;

}

console.log(sumInRange(8, 10)); // Вывод: 27

console.log(sumInRange(12, 15)); // Вывод: 54

console.log(sumInRange(0, 3)); // Вывод: 6

4) Напишите функцию, которая получает массив чисел и возвращает среднее ариф. значение

function averageValue(numbers) {

if (numbers.length == 0) {

return 0; }

let sum = 0;

for (let i = 0; i < numbers.length; i++) {

sum += numbers[i];

}

return sum / numbers.length;

}

let nums = [2, 4, 6, 8, 10];

console.log(averageValue(nums)); // Вывод: 6

let emptyArray = [];

console.log(averageValue(emptyArray)); // Вывод: 0

5) Напишите функцию, которая конвертирует градусы из шкалы по Цельсию в градусы в шкалу по Фаренгейту

function celsiusToFahrenheit(celsius) {

let fahrenheit = (celsius * 9 / 5) + 32;

return fahrenheit;

}

let celsiusTemp = 25;

let fahrenheitTemp = celsiusToFahrenheit(celsiusTemp);

console.log(fahrenheitTemp); // Вывод: 77
